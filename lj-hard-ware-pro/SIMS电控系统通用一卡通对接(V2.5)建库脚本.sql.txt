--////////////////////////////////////////////////////////////////////////////////////
--向临时表写入购电数据
--////////////////////////////////////////////////////////////////////////////////////
CREATE PROCEDURE SP_KDDB_POSTFEE
    @RSULT        int OUTPUT,
    @B_ID           int,
    @XQ_ID        int,
    @R_ID          int,
    @FEE           float,
    @DT            datetime,
   @CSN          char(20)
AS
BEGIN
    DECLARE @TR int
    SET @TR=0
    BEGIN TRANSACTION
        INSERT KD_TMP     
            VALUES (@B_ID, @XQ_ID, @R_ID, @FEE, @DT, @CSN)
        SET @TR=1
    COMMIT TRANSACTION
    SET @RSULT=@TR
END
GO
